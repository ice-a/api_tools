请求方法:

import requests
import json

url = "http://127.0.0.1:5000/generate_wav"

payload = json.dumps({
   "text": "你好，我是李大山",
   "speaker_voice": "zh-TW-HsiaoChenNeural"
})
headers = {
   'User-Agent': 'Apifox/1.0.0 (https://apifox.com)',
   'Content-Type': 'application/json'
}
response = requests.request("POST", url, headers=headers, data=payload)
print(response.text)
